## firebase realtime database kurallar

Kullanıcının yalnızca kendi uid değerine sahip olduğu dizine erişebilmesi amaçlanıyor.

Aşağıdaki kurallar ile soru kaydı başarılı oldu.
Hasılat ekranında grafik verileri gösterimi başarısız oldu.
Konu çalışma süresi kayıt ekleme denenmedi.

*kaynak: https://firebase.google.com/docs/rules/basics?hl=en*

```json
{
  "rules": {
    "users":{
      "$uid":{
        // eğer kullanıcı id değeri kayıt başlığındaki id değeri ile aynı ise okuma izni ver
        ".read": "auth.uid==$uid",
        // eğer kullanıcı id değeri kayıt başlığındaki id değeri ile aynı ise yazma izni ver
        ".write": "auth.uid==$uid"
      }
    }
  }
}
```

## Firebase Veri İşleme Örnekleri

Firebase üzerinden çekilen verilerin işlenmesinde örnek alınabilecek satırlar.

```js
/* soru kaydı ekleme */
$("#addQuestionBtn").click(function () {
    var lesson = $('#lesson').val()
    var questionCount = $('#questionCount').val()
    firebase.database().ref().child("users").child(petek.veri.key).child("records").push(
        {
            lesson: lesson,
            count: questionCount,
            time: Date.now()
        }
    );
    alert("Soru Kaydı Eklendi 👍")
    $("#questionCount").val('');
});

/* veri ekleme */
$("body").on("change", ".button", function () {
    var $completed = $(this).prop("checked");
    var $key = $(this).data("key");
    firebase.database().ref("users/" + petek.veri.key).child("todos").child($key).child("completed").set($completed);
})

/* profil bilgileri güncelleme */
$("#saveProfileBtn").click(function () {
    var name = $('#name').val()
    var surname = $('#surname').val()
    var city = $('#city').val()
    var grade = $('#grade').val()
    var birthdate = $('#birthdate').val()
    firebase.database().ref().child("users").child(petek.veri.key).update(
        {
            name: name,
            surname: surname,
            city: city,
            grade: grade,
            birthdate: birthdate
        }
    );
    alert("Bilgiler Güncellendi 👍")
});

/* veri silme */
$("body").on("click", ".removeBtn", function () {
    var $key = $(this).data("key");
    firebase.database().ref("users/" + petek.veri.key).child("records").child($key).remove();
})

/* veri çekme ve işleme */
var userRef = firebase.database().ref().child("users/" + petek.veri.key);
userRef.on("value", function (snapshot) {
    if (snapshot.val()) {
        $('#name').val(snapshot.val().name)
        $('#surname').val(snapshot.val().surname)
        $('#city').val(snapshot.val().city)
        $('#grade').val(snapshot.val().grade)
        $('#birthdate').val(snapshot.val().birthdate)
    }
})

/* oturum kapatma */
$("#logout").click(function () {
    firebase.auth().signOut()
        .then(function () {
            window.location.href = "giris-yap.html";
        })
})
```
